108 inline std::string FormatCommand(std::string_view tmpl, const std::unordered_map<std::string, std::string>& values) {
109  std::string result(tmpl);
110  for (const auto& [key, val] : values) {
111  std::string placeholder = "{" + key + "}";
112  size_t pos = 0;
113  while ((pos = result.find(placeholder, pos)) != std::string::npos) {
114  result.replace(pos, placeholder.length(), val);
115  pos += val.length();
116  }
117  }
118  return result;
119 }
